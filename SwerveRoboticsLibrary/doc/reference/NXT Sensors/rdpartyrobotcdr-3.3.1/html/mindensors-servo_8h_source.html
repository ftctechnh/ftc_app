00105   memset(NXTSERVO_I2CRequest, 0, sizeof(tByteArray));

00107   NXTSERVO_I2CRequest[1] = address;
00108   NXTSERVO_I2CRequest[2] = NXTSERVO_SPEED_CHAN1 + (servochan - 1);
